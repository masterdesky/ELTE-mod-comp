\begin{alertblock}{Processes, Threads}
	\begin{itemize}
		\item Process
		\begin{itemize}
			\item Name for a single, running program
		\end{itemize}
		\item Thread
		\begin{itemize}
			\item A running sequence of operations 
			\item A process can be broken down into several threads
			\item They always run \q{in parallel}
		\end{itemize}
	\end{itemize}
